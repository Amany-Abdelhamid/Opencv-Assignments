def my_gradient(img, g_filter):
    [si,sj]=img.shape
    out=np.zeros((si,sj))
    img=np.pad(img, (1,1), 'constant', constant_values=(0,0))
    for i in range(0,si):
        for j in range(0,sj):
            out[i,j]=(g_filter[0,0]*img[i-1,j-1] +
                      g_filter[0,1]*img[i-1,j]+
                      g_filter[0,2]*img[i-1,j+1] +
                      
                      g_filter[1,0]*img[i,j-1] +
                      g_filter[1,1]*img[i,j]+
                      g_filter[1,2]*img[i,j+1] +
                      
                      g_filter[2,0]*img[i+1,j-1] +
                      g_filter[2,1]*img[i+1,j] +
                      g_filter[2,2]*img[i+1,j+1]) /9
    return out

def myCanny(im): 
    # Sobel operator 
        op1 = np.array([[-1, 0, 1], 
                  [-2, 0, 2], 
                  [-1, 0, 1]]) 
        op2 = np.array([[-1, -2, -1], 
                  [ 0,  0,  0], 
                  [ 1,  2,  1]]) 
        kernel1 = np.zeros(im.shape) 
        kernel1[:op1.shape[0], :op1.shape[1]] = op1 
        kernel1 = np.fft.fft2(kernel1) 

        kernel2 = np.zeros(im.shape) 
        kernel2[:op2.shape[0], :op2.shape[1]] = op2 
        kernel2 = np.fft.fft2(kernel2) 


        fim = np.fft.fft2(im) 
        Gx = np.real(np.fft.ifft2(kernel1 * fim)).astype(float) 
        Gy = np.real(np.fft.ifft2(kernel2 * fim)).astype(float) 


        G = np.sqrt(Gx**2 + Gy**2) 
        Theta = np.arctan(Gy, Gx) * 180 / np.pi 
        maxi = np.maximum(G, Theta)
        Threshold = maxi > 120
        return Threshold

def myhough_line(img): 
    
    # Rho and Theta ranges 
    thetas = np.deg2rad(np.arange(-90.0, 90.0)) 
    width, height = img.shape 
    diag_len = int(round(np.sqrt(width * width + height * height))) 
    rhos = np.linspace(-diag_len, diag_len, diag_len * 2) 


    #Compute sin and cos
    cos_theta = np.cos(thetas) 
    sin_theta = np.sin(thetas) 
    num_thetas = len(thetas) 


    # H(theta, rho) initialization 
    H = np.zeros((2 * diag_len, num_thetas), dtype=np.uint8) 
    # edges of image 
    y_idxs, x_idxs = np.nonzero(img) 


    # Vote in the hough accumulator 
    for i in range(len(x_idxs)): 
        x = x_idxs[i] 
        y = y_idxs[i] 
        for theta_idx in range(num_thetas): 
        # Calculate rho.  
            rho = diag_len + int(round(x * cos_theta[theta_idx] + y * sin_theta[theta_idx])) 
            H[rho, theta_idx] += 1 
      
    
    #get maximum H(theta,rho)
    '''lines=[]
    max_idxs = np.argwhere(accumulator == np.amax(accumulator))
    for i in range(len(max_idxs)): 
        rho = rhos[i] 
        theta = thetas[i] 
        lines.append([rho,theta])'''
    max_idx = np.argmax(H)
    rho = rhos[int(max_idx / H.shape[1])]
    theta = thetas[int(max_idx % H.shape[1])]
    
    
    return  rho,theta
